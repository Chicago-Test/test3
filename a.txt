#cls
cd $PSScriptRoot

Remove-Variable * -ErrorAction SilentlyContinue

$CREATE_DUMMY_FILES = $false
$path1 = "E:\temp\LibXDiff-call\powershell\1\"
$path2 = "E:\temp\LibXDiff-call\powershell\2"

$path1 = $path1.TrimEnd("\");
$path2 = $path2.TrimEnd("\");

if (![System.IO.Directory]::Exists($path1) -or ![System.IO.Directory]::Exists($path2)) {
    #Write-Host "Folder does not exist"
    Exit
}

#Delete size zero files
#Get-ChildItem -Path $path1 -Recurse -File | Where-Object { $_.Length -eq 0 } | Remove-Item -Force
#Get-ChildItem -Path $path2 -Recurse -File | Where-Object { $_.Length -eq 0 } | Remove-Item -Force

$list1 = (Get-ChildItem $path1 -File -Recurse | % { $x = $_.FullName; $x = $x.Substring($path1.Length, $x.Length - $path1.Length); return $x; })
$list2 = (Get-ChildItem $path2 -File -Recurse | % { $x = $_.FullName; $x = $x.Substring($path2.Length, $x.Length - $path2.Length); return $x; })

$ListOfFilesNotExist = Compare-Object -ReferenceObject $list1 -DifferenceObject $list2

$strBuilder = [System.Text.StringBuilder]"";

foreach ($c in $ListOfFilesNotExist) {
    if ($c.SideIndicator -eq "=>") {
        [void]$strBuilder.AppendLine($path1 + $c.InputObject);
    }
    elseif ($c.SideIndicator -eq "<=") {
        [void]$strBuilder.AppendLine($path2 + $c.InputObject);
    }
}
$strBuilder.ToString() | Out-File -Encoding "UTF8" "tmp_file_list.csv"
$i = 0;
if ($CREATE_DUMMY_FILES -eq $true) {
    if ($strBuilder.Length -gt 0) {
        $lines = $strBuilder.ToString() -split "`n"
        foreach ($c in $lines) {
            if ($c.Length -gt 0) { $x = (New-Item -ItemType File -Path $c.Trim()); $i++; }
        }
    }
    Write-Host "Created " $i "dummy files"
}
