        $ChromeOptions.AddUserProfilePreference("profile.default_content_setting_values.images", 2) # no images
    
        $ChromeDriver = New-Object OpenQA.Selenium.Chrome.ChromeDriver($ChromeOptions)
        $url = $baseURL
        #$url = "https://tipswatch.com/2023/08/27/oh-no-ive-become-a-treasury-market-day-trader/"

        $strBuilder = [System.Text.StringBuilder]""
        [void]$strBuilder.Append([ClassTipsWatch]::title + "<br>----------------<br>")
    
        $elem = $ChromeDriver.FindElements( $xlDirectionBy::xpath("//div[@class='entry-content']"))
        $x1 = $elem[0].GetAttribute("innerHTML")
        [void]$strBuilder.Append($x1)
        # Get comments
        try {
            $x2 = $ChromeDriver.FindElements( $xlDirectionBy::xpath("//*[@class='commentlist']/li[starts-with(@id,'li-comment-')]"))
            foreach ($item in $x2) {
                [void]$strBuilder.Append( $item.GetAttribute("innerHTML"))
                #[void]$strBuilder.Append("<br>--------------------<br> >")
                #$strBuilder.Length -= 1 # remove last ">"
            }
        }
        catch {
            break            
        }
    
        $ChromeDriver.close()
        $ChromeDriver.quit()

        # Remove image, ads, etc. from HTML
        $Parser = New-Object AngleSharp.Html.Parser.HtmlParser
        $Parsed = $Parser.ParseDocument($strBuilder.ToString())
        foreach ($c in $Parsed.QuerySelectorAll("img,iframe,div[id='content_btf_wrapper'],div[class*='sharedaddy']")) {
            $c.Remove();
        }

        return $Parsed.Body.innerHtml # "$Parsed.Source.Text" does not being affected by Remove().
    }
}

function Parse_DoctorOfCredit_HTMLfunc {
    param (
        [string]$Source
    )
    $Parser = New-Object AngleSharp.Html.Parser.HtmlParser
    $Parsed = $Parser.ParseDocument($Source)
    $elements1 = $Parsed.QuerySelector("[class='entry-title']")
    $global:title = $elements1.TextContent
    $x2 = $global:title + "<br>"

    $elem = $Parsed.GetElementById("main")
    #var selector = "img.clipboardData.b.c"; //tag img with class clipboardData, class b, and class c
    foreach ($c in $elem.QuerySelectorAll("div.code-block")) {
        # remove <img, <iframe, class including ...
        $c.Remove();
    }
    $x1 = $elem.GetElementsByClassName("entry-content")
    $x2 += $x1.OuterHtml

    $x2 += "<br>Comments<br>"
    # $x1 = $Parsed.GetElementById("wcThreadWrapper")
    # $x1 = $x1.QuerySelectorAll("div.wc_comment_level-1")
    $x1 = $Parsed.GetElementsByClassName("wpd-thread-list")
    $x1 = $x1.QuerySelectorAll("div[class*='wpd_comment_level-1']")
    foreach ($y in $x1) {
        $x2 += "<br>===========================<br>"
        # $x1a = $y.GetElementsByClassName("wc-comment-text")
        $x1a = $y.GetElementsByClassName("wpd-comment-text")
        foreach ($y1 in $x1a) {
            $tmp = "<br>" + "-------------------" + "<br>"
            $x2 += ($y1.TextContent + $tmp)
        }
        $x2 = $x2.Substring(0, $x2.Length - $tmp.Length) # remove last "---------"
    }
    $x2
    return
}
