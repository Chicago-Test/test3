---- Create pfx ------
$cert=New-SelfSignedCertificate -Type DocumentEncryptionCert -Subject "CN=test" -CertStoreLocation "Cert:\CurrentUser\My" -KeyExportPolicy Exportable -NotAfter (Get-Date).AddYears(1)
$cert.Thumbprint
$MyPwd=ConvertTo-SecureString -String "a" -Force -AsPlainText
Export-PfxCertificate -Cert $cert -Password $MyPwd -FilePath "e:\self-signed.pfx"
----- Encrypt & Decrypt -------
$enc = ("This is a secret message" | Protect-CmsMessage -To cn=test)
$enc = ("This is a secret message" | Protect-CmsMessage -To "CN=test","CN=test1")  # multiple recipients
$enc | Unprotect-CmsMessage

-------import(load) pfx -----------
# $cert = Get-PfxCertificate -FilePath "e:\self-signed.pfx"  (ask for password)
$cert = New-Object System.Security.Cryptography.X509Certificates.X509Certificate2
$cert.Import("e:\self-signed.pfx",'a','DefaultKeySet')
$cert.HasPrivateKey
--------- export only cert ---------
###  $tmp = Get-PfxData -FilePath "e:\self-signed.pfx" -Password $MyPwd
Export-Certificate -Cert $cert -FilePath "e:\self-signed.cer" -Type Cert
$cert.Import("e:\self-signed.cer")
$cert.HasPrivateKey
------ Encrypt with cert -------
Protect-CmsMessage -Content "Your Secret Message" -To "e:\self-signed.cer"
Protect-CmsMessage -Content "Your Secret Message" -To "e:\self-signed.cer","e:\self-signed2.cer"

------ Convert binary pfx to text(base64) -------
$PFX_FILE = get-content "e:\self-signed.pfx" -Encoding Byte
$out1=[System.Convert]::ToBase64String($PFX_FILE)
------ Load text(base64) to certification object -------
$bytes = [Convert]::FromBase64String($out1)
$cert = New-Object System.Security.Cryptography.X509Certificates.X509Certificate2
$cert.Import($bytes,'a','DefaultKeySet')
